<NpmInstall packageName="@zenbu-ui/tabs" />

## Usage

```jsx
import { Tabs } from '@zenbu-ui/tabs'
import { Text } from '@zenbu-ui/typography'

const App = () => {
  return(
    <Tabs>
      <Tabs.Item title={(<Text span fontSize="sm">Account</Text>)}>
        ...
      </Tabs.Item>

      <Tabs.Item title={(<Text span fontSize="sm">Password</Text>)}>
        ...
      </Tabs.Item>
    </Tabs>
  )
}

export default App
```
