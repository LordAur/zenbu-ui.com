## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/input" />

## Usage

```jsx
import { Input } from '@zenbu-ui/input'

const App = () => {
  return(
    <Input.File name="tester" accept="*" />
  )
}

export default App
```

### Text State
<PreviewInputFileText />

```jsx
<Input.File name="tester" accept="*"
text="Pilih Data"
textState="Tidak ada data yang dipilih" />
```

### Multiple
<PreviewInputFileMultiple />

```jsx
<Input.File name="tester" accept="*" multiple />
```

### Error
<PreviewInputFileError />

```jsx
<Input.File
name="tester" accept="*"
maxSize={1000}
errorMaxSize={(maxSize: string) => `File tidak boleh lebih besar dari ${maxSize}`} />
```
