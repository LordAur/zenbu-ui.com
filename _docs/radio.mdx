## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/input" />

## Usage

```jsx
import { Input } from '@zenbu-ui/input'

const App = () => {
  return(
    <>
      <Input.Radio checked
      value={1} name="tester"
      label={(
        <Text span fontSize="sm">
          Default checkbox 1
        </Text>
      )} />

      <Input.Radio
      value={2} name="tester"
      label={(
        <Text span fontSize="sm">
          Default checkbox 2
        </Text>
      )} />
    </>
  )
}

export default App
```
