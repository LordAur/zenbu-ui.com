## Installation

<NpmInstall packageName="@zenbu-ui/accordion" />

## Usage
```jsx
import { Accordion } from '@zenbu-ui/accordion'

const App = () => {
  return(
    <Accordion>
      <Accordion.Item title="Accordion #1">...</Accordion.Item>
      <Accordion.Item title="Accordion #2">...</Accordion.Item>
      <Accordion.Item title="Accordion #3">...</Accordion.Item>
    </Accordion>
  )
}

export default App
```
