## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/table" />

## Usage
Table components used array objects to create columns and rows. See the key and type below:

### Columns Structure
<ColumnTable />

```jsx
import { Table } from '@zenbu-ui/table'

const columns = [{
  header: "Name",
  accessor: "name",
  position: "left"
}, {
  header: "Job",
  accessor: "job",
  position: "left"
}, {
  header: "Age",
  accessor: "age",
  position: "left",
  width: "12"
}]

const rows = [{
  name: "Jhon",
  job: "Frontend Engineer",
  age: 24
}, {
  name: "Gill",
  job: "DevOps",
  age: 31
}, {
  name: "Oesman",
  job: "Backend Engineer",
  age: <b>27</b>
}]

const App = () => {
  return(
    <Table columns={columns} rows={rows} />
  )
}

export default App
```

### Checkbox
<PreviewTableCheckbox />

```jsx
import { useState } from 'react'
import { Table } from '@zenbu-ui/table'

const [selected, setSelected] = useState([])

const App = () => {
  return(
    <Table stripe checkbox selected={selected}
    columns={columns} rows={rows}
    textColor="gray" textColorContrast="700"
    borderColor="gray" borderColorContrast="300"
    colColor="gray" colColorContrast="400"
    rowColor="gray" rowColorContrast="200"
    stripeColor="gray" stripeColorContrast="300"
    onSelected={(selected) => setSelected([...selected])} />
  )
}
```
