## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/typography" />

## Usage

```jsx
import { Blockquote } from '@zenbu-ui/blockquote'

const App = () => {
  return(
    <Blockquote width="96" textAlign="center"
    cite="https://...">
      ...
    </Blockquote>
  )
}

export default App
```
