## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/skeleton" />

## Usage

```jsx
import { Skeleton } from '@zenbu-ui/skeleton'

const App = () => {
  return(
    <>
      <Skeleton width="20" mr="2" />
      <Skeleton width="44" mr="2" />
      <Skeleton width="72" />
    </>
  )
}

export default App
```
