## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/input" />

## Usage

```jsx
import { Input } from '@zenbu-ui/input'

const App = () => {
  return(
    <Input.Checkbox name="tester" l
    label={(<Text span fontSize="sm">Default checkbox</Text>)} />
  )
}

export default App
```
