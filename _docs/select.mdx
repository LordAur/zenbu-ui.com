## Usage

```jsx
import { Input } from '@zenbu-ui/input'

const App = () => {
  const data = [{
    key: "apple",
    value: "apple",
    text: "Apple"
  }, {
    key: "orange",
    value: "orange",
    text: "Orange"
  }, {
    key: "mango",
    value: "mango",
    text: "Mango"
  }]

  return(
    <Input.Select name="tester" data={data} placeholder="Choose fruit" />
  )
}

export default App
```
