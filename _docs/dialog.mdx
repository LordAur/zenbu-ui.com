## Usage
```jsx
import { useState } from 'react'
import { Dialog } from '@zenbu-ui/dialog'

const App = () => {
  const [visible, setVisible] = useState(false)

  return(
    <>
      <Button onClick={() => setVisible(true)}>Show Dialog</Button>

      <Dialog visible={visible} onClose={() => setVisible(false)}>
        ...
      </Dialog>
    </>
  )
}

export default App
```
