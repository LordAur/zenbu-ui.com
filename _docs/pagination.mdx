## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/pagination" />

## Usage

```jsx
import { Pagination } from '@zenbu-ui/pagination'

const App = () => {
  return(
    <Pagination total={100} />
  )
}
```

### Simple

<PreviewPaginationSimple />

```jsx
<Pagination total={100} border={false}
color="transparent"
colorHover="slate" colorHoverContrast="300"
darkColorHover="slate" darkColorHoverContrast="500"
textColor="slate" textColorContrast="700"
darkTextColor="slate" darkTextColorContrast="200"/>
```

### Show Total Data
<PreviewPaginationTotal />

```jsx
<Pagination total={100}
showTotal={(total) => (
  <Text
  color="slate" colorContrast="700"
  darkColor="slate" darkColorContrast="200"
  fontSize="sm">
    Total {total} data
  </Text>
)} />
```

### Show Range Data
<PreviewPaginationRange />

```jsx
<Pagination total={100}
showTotal={(total, range) => (
  <Text
  color="slate" colorContrast="700"
  darkColor="slate" darkColorContrast="200"
  fontSize="sm">
    {range[0]}-{range[1]} of {total} items
  </Text>
)} />
```
