## Installation
Install the component from your command line.

<NpmInstall packageName="@zenbu-ui/list" />

## Usage
```jsx
import { List } from '@zenbu-ui/list'

const App = () => {
  return(
    <List>
      <List.Item>Apples</List.Item>
      <List.Item>Pears</List.Item>
      <List.Item>Oranges</List.Item>
    </List>
  )
}

export default App
```
