## Installation

<NpmInstall packageName="@zenbu-ui/tags" />

## Usage

```jsx
import { Tag } from '@zenbu-ui/tags'

const App = () => {
  return(
    <Tag
    color="blue" colorContrast="500"
    textColor="white">
      <Icon icon="mail-open" height="4" mr="1" />
      Mail
    </Tag>
  )
}

export default App
```
